WebConfig文件，相当于前端控制器，替代dispatcher-servlet.xml文件 

@Configuration
@EnableWebMvc
@ComponentScan(basePackages = "com.xujn")
@PropertySource(value = "classpath:freemarker.properties",encoding = "UTF-8")
public class WebConfig extends WebMvcConfigurerAdapter &#123;

@Autowired
    private Environment env;
    //    视图解析器
    @Bean
    public ViewResolver viewResolver()&#123;
        InternalResourceViewResolver resolver =
                new InternalResourceViewResolver();
        resolver.setViewClass(JstlView.class);
        resolver.setPrefix("/WEB-INF/views/");
        resolver.setSuffix(".jsp");
        resolver.setOrder(1);
        resolver.setExposeContextBeansAsAttributes(true);
        return resolver;
    &#125;
//    freemarker解析器
    @Bean
    public FreeMarkerViewResolver FreeMarkerViewResolver()&#123;
        FreeMarkerViewResolver resolver = new FreeMarkerViewResolver("",".ftl");
        resolver.setCache(true);
        resolver.setContentType("text/html;charset=UTF-8");
        resolver.setOrder(0);
        return resolver;
    &#125;
        //    配置资源映射
    @Override
    public void addResourceHandlers(ResourceHandlerRegistry registry) &#123;
//        SpringMVCBlog/src/main/resources
        registry.addResourceHandler("/resources/**").
                addResourceLocations("classpath:/resources/").setCachePeriod(2592000);
    &#125;
    //    json解析
    @Bean
    public MappingJackson2HttpMessageConverter mappingJackson2HttpMessageConverter() &#123;
        MappingJackson2HttpMessageConverter converter = new MappingJackson2HttpMessageConverter();
        List<MediaType> supportedMediaTypes = new ArrayList<>();
        supportedMediaTypes.add(MediaType.APPLICATION_JSON_UTF8);
        supportedMediaTypes.add(MediaType.TEXT_HTML);
        converter.setSupportedMediaTypes(supportedMediaTypes);
        //    converter.setObjectMapper(objectMapper());
        return converter;
    &#125;
    // 主题资源配置
    @Bean(name = "themeSource")
    public ThemeSource resourceBundleThemeSource() &#123;
        ResourceBundleThemeSource source = new ResourceBundleThemeSource();
        source.setDefaultEncoding("UTF-8");
        source.setBasenamePrefix("themes.");
        return source;
    &#125;

    @Bean
    public CookieThemeResolver themeResolver() &#123;
        CookieThemeResolver resolver = new CookieThemeResolver();
        resolver.setDefaultThemeName("anime");
        resolver.setCookieName("my-theme-cookie");
        return resolver;
    &#125;

    @Bean
    public ThemeChangeInterceptor themeChangeInterceptor() &#123;
        ThemeChangeInterceptor interceptor = new ThemeChangeInterceptor();
        interceptor.setParamName("theme");
        return interceptor;
    &#125;

    @Override
    public void addInterceptors(InterceptorRegistry registry) &#123;
        registry.addInterceptor(themeChangeInterceptor());
    &#125;
    @Bean(name = "multipartResolver")
    public CommonsMultipartResolver commonsMultipartResolver()&#123;
        CommonsMultipartResolver resolver = new CommonsMultipartResolver();
        resolver.setMaxUploadSize(5 * 1024 * 1024);
        resolver.setDefaultEncoding("UTF-8");
        return resolver;
    &#125;

@Configuration
public class DataSourceConfig &#123;

    @Bean( name = "dataSource")

        BasicDataSource ds = new BasicDataSource();
        ds.setDriverClassName("com.mysql.jdbc.Driver");
        ds.setUrl("jdbc:mysql://localhost:3306/blog?" + "useUnicode=true&characterEncoding=UTF-8");
        ds.setUsername("username");
        ds.setPassword("password");
        ds.setInitialSize(5);
        ds.setMaxTotal(10);
